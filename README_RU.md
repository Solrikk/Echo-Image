async def process_image(session, image_entry, target_image):
  try:
    # –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è –∑–∞–≥—Ä—É–∑–∫–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –ø–æ —É–∫–∞–∑–∞–Ω–Ω–æ–º—É URL.

                       max(current_image.shape[:2]))

    # –ò–∑–º–µ–Ω–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞ –æ–±–æ–∏—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –¥–æ —Ä–∞—Å—Å—á–∏—Ç–∞–Ω–Ω–æ–≥–æ –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞ –¥–ª—è –µ–¥–∏–Ω–æ–æ–±—Ä–∞–∑–Ω–æ–≥–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è.

                                      (optimal_size, optimal_size))
    current_image_resized = cv2.resize(current_image, 
                                       (optimal_size, optimal_size))
    # –ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è –æ–±–æ–∏—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –≤ –≥—Ä–∞–¥–∞—Ü–∏–∏ —Å–µ—Ä–æ–≥–æ –¥–ª—è —É–ø—Ä–æ—â–µ–Ω–∏—è –ø–æ—Å–ª–µ–¥—É—é—â–∏—Ö —Ä–∞—Å—á–µ—Ç–æ–≤.

    current_gray = cv2.cvtColor(current_image_resized, cv2.COLOR_BGR2GRAY)

    current_keypoints, current_descriptors = orb.detectAndCompute(current_gray, None)
    # –ï—Å–ª–∏ –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä—ã –Ω–µ –Ω–∞–π–¥–µ–Ω—ã, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Å—Ä–∞–≤–Ω–µ–Ω–∏—è —Å—Ö–æ–¥—Å—Ç–≤–∞ —Ä–∞–≤–Ω—ã–π 0.

    # –ü–∞—Ä–∞–º–µ—Ç—Ä—ã –¥–ª—è —Å–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏—è FLANN –¥–ª—è –ø–æ–∏—Å–∫–∞ –ª—É—á—à–∏—Ö —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π –º–µ–∂–¥—É –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä–∞–º–∏.
    index_params = dict(algorithm=6,
                        table_number=6,
                        key_size=12,
                        multi_probe_level=1)
    search_params = dict(checks=50)  # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ–≤–µ—Ä–æ–∫ –¥–ª—è —Å–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏—è
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å–æ–ø–æ—Å—Ç–∞–≤–∏—Ç–µ–ª—è FLANN —Å —É–∫–∞–∑–∞–Ω–Ω—ã–º–∏ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏.
    flann = cv2.FlannBasedMatcher(index_params, search_params)
