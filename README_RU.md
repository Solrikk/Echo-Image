```Python 
      # –†–∞—Å—á–µ—Ç –∏–Ω–¥–µ–∫—Å–∞ —Å—Ç—Ä—É–∫—Ç—É—Ä–Ω–æ–≥–æ —Å—Ö–æ–¥—Å—Ç–≤–∞ (SSIM) –º–µ–∂–¥—É –¥–≤—É–º—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏.
      ssim_index = ssim(target_gray, current_gray)
      # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –¥–µ—Ç–µ–∫—Ç–æ—Ä–∞ ORB –¥–ª—è –∏–∑–≤–ª–µ—á–µ–Ω–∏—è –ø—Ä–∏–∑–Ω–∞–∫–æ–≤.
      orb = cv2.ORB_create(nfeatures=500)
      # –í—ã—è–≤–ª–µ–Ω–∏–µ –∫–ª—é—á–µ–≤—ã—Ö —Ç–æ—á–µ–∫ –∏ –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä–æ–≤ –¥–ª—è –æ–±–æ–∏—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π.
      target_keypoints, target_descriptors = orb.detectAndCompute(
          target_gray, None)
      current_keypoints, current_descriptors = orb.detectAndCompute(
          current_gray, None)
      # –í–æ–∑–≤—Ä–∞—Ç —Ä–∞–Ω–µ–µ, –µ—Å–ª–∏ –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä—ã –Ω–µ –Ω–∞–π–¥–µ–Ω—ã –Ω–∏ –≤ –æ–¥–Ω–æ–º –∏–∑ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π.
      if target_descriptors is None or current_descriptors is None:
        return (0, image_entry["url"])
      # –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –¥–ª—è –º–∞—Ç—á–µ—Ä–∞ –Ω–∞ –æ—Å–Ω–æ–≤–µ FLANN, –∏—Å–ø–æ–ª—å–∑—É–µ–º–æ–≥–æ –¥–ª—è –ø–æ–∏—Å–∫–∞ —Ö–æ—Ä–æ—à–∏—Ö —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π.
      index_params = dict(algorithm=6,
                          table_number=6,
                          key_size=12,
                          multi_probe_level=1)
      search_params = dict(checks=50)
      flann = cv2.FlannBasedMatcher(index_params, search_params)
      # –°–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏–µ –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä–æ–≤ –º–µ–∂–¥—É –¥–≤—É–º—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏ –∏ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è —Ö–æ—Ä–æ—à–∏—Ö —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π.
      matches = flann.knnMatch(target_descriptors, current_descriptors, k=2)
      good_matches = [m for m, n in matches if m.distance < 0.75 * n.distance]
      # –†–∞—Å—á–µ—Ç –æ—Ü–µ–Ω–∫–∏ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ –Ω–∞ –æ—Å–Ω–æ–≤–µ —Ö–æ—Ä–æ—à–∏—Ö —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π.
      feature_score = len(good_matches) / float(len(target_keypoints))
      # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –≥–∏—Å—Ç–æ–≥—Ä–∞–º–º –¥–ª—è –æ–±–æ–∏—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –≤ –∫–∞–Ω–∞–ª–∞—Ö RGB.
      target_hist = cv2.calcHist([target_image_resized], [0, 1, 2], None,
                                 [32, 32, 32], [0, 256, 0, 256, 0, 256])
      current_hist = cv2.calcHist([current_image_resized], [0, 1, 2], None,
                                  [32, 32, 32], [0, 256, 0, 256, 0, 256])
      # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –≥–∏—Å—Ç–æ–≥—Ä–∞–º–º.
      cv2.normalize(target_hist, target_hist)
      cv2.normalize(current_hist, current_hist)
      # –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –≥–∏—Å—Ç–æ–≥—Ä–∞–º–º —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –º–µ—Ç–æ–¥–∞ –∫–æ—Ä—Ä–µ–ª—è—Ü–∏–∏.
      hist_score = cv2.compareHist(target_hist, current_hist,
                                   cv2.HISTCMP_CORREL)
      # –†–∞—Å—á–µ—Ç –æ–∫–æ–Ω—á–∞—Ç–µ–ª—å–Ω–æ–π –æ—Ü–µ–Ω–∫–∏ –ø—É—Ç–µ–º —É—Å—Ä–µ–¥–Ω–µ–Ω–∏—è –æ—Ü–µ–Ω–æ–∫ SSIM, –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ –∏ –≥–∏—Å—Ç–æ–≥—Ä–∞–º–º.
      final_score = (feature_score + ssim_index + hist_score) / 3
      return (final_score, image_entry["url"])
  except Exception as e:
    # –û–±—Ä–∞–±–æ—Ç–∫–∞ –ª—é–±—ã—Ö –æ—à–∏–±–æ–∫ –≤–æ –≤—Ä–µ–º—è –ø—Ä–æ—Ü–µ—Å—Å–∞ –∏ –≤–æ–∑–≤—Ä–∞—Ç –Ω—É–ª–µ–≤–æ–π –æ—Ü–µ–Ω–∫–∏.
    print(f"–ù–µ —É–¥–∞–ª–æ—Å—å –æ–±—Ä–∞–±–æ—Ç–∞—Ç—å –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ {image_entry['url']} –∏–∑-–∑–∞ {e}")
    return (0, image_entry["url"])
```

## _ORB (Oriented FAST and Rotated BRIEF)_ 
ORB –º–µ—Ç–æ–¥, –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã–π –≤ –∫–æ–º–ø—å—é—Ç–µ—Ä–Ω–æ–º –∑—Ä–µ–Ω–∏–∏, –æ—Å–æ–±–µ–Ω–Ω–æ –ø–æ–ø—É–ª—è—Ä–Ω—ã–π –¥–ª—è –∑–∞–¥–∞—á, —Å–≤—è–∑–∞–Ω–Ω—ã—Ö —Å —Ä–∞—Å–ø–æ–∑–Ω–∞–≤–∞–Ω–∏–µ–º –æ–±—ä–µ–∫—Ç–æ–≤, —Å–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –∏ –æ—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏–µ–º. –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ —Å–æ—Å—Ä–µ–¥–æ—Ç–æ—á–µ–Ω –Ω–∞ –±—ã—Å—Ç—Ä–æ–º –ø–æ–∏—Å–∫–µ –∫–ª—é—á–µ–≤—ã—Ö —Ç–æ—á–µ–∫ –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è—Ö –∏ –æ–ø–∏—Å–∞–Ω–∏–∏ –∏—Ö —Ç–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º, —á—Ç–æ–±—ã –º–æ–∂–Ω–æ –±—ã–ª–æ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ —Å—Ä–∞–≤–Ω–∏–≤–∞—Ç—å –∏—Ö. ORB –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç –±—ã—Å—Ç—Ä—ã–π –∏ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω—ã–π —Å–ø–æ—Å–æ–± —Å–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π, –Ω–µ—Å–º–æ—Ç—Ä—è –Ω–∞ –∏–∑–º–µ–Ω–µ–Ω–∏—è –≤ —É–≥–ª–µ –æ–±–∑–æ—Ä–∞, –º–∞—Å—à—Ç–∞–±–µ –∏–ª–∏ –æ—Å–≤–µ—â–µ–Ω–∏–∏.

1) **Oriented FAST (Features from Accelerated Segment Test):** –≠—Ç–æ—Ç –∫–æ–º–ø–æ–Ω–µ–Ω—Ç –æ—Ç–≤–µ—á–∞–µ—Ç –∑–∞ –æ–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ —Ç–æ—á–µ–∫ –∏–Ω—Ç–µ—Ä–µ—Å–∞ (–∏–ª–∏ –∫–ª—é—á–µ–≤—ã—Ö —Ç–æ—á–µ–∫) –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–∏. –û–Ω –±—ã—Å—Ç—Ä–æ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç —É–≥–ª—ã –∏–ª–∏ –∫—Ä–∞—è, –∫–æ—Ç–æ—Ä—ã–µ –≤—ã–¥–µ–ª—è—é—Ç—Å—è –ø–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—é —Å –æ–∫—Ä—É–∂–∞—é—â–∏–º–∏ –æ–±–ª–∞—Å—Ç—è–º–∏. –¢–∞–∫–∏–º –æ–±—Ä–∞–∑–æ–º, –º–æ–∂–Ω–æ –∏–¥–µ–Ω—Ç–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞—Ç—å –∑–Ω–∞—á–∏—Ç–µ–ª—å–Ω—ã–µ –∏–ª–∏ —É–Ω–∏–∫–∞–ª—å–Ω—ã–µ —Å–µ–∫—Ü–∏–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.

2) **Rotated BRIEF (Binary Robust Independent Elementary Features):** –ü–æ—Å–ª–µ –Ω–∞—Ö–æ–∂–¥–µ–Ω–∏—è –∫–ª—é—á–µ–≤—ã—Ö —Ç–æ—á–µ–∫ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ —Å–æ–∑–¥–∞—Ç—å –æ–ø–∏—Å–∞–Ω–∏–µ –∫–∞–∂–¥–æ–π –∏–∑ –Ω–∏—Ö –¥–ª—è –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–∏ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è —Å –∫–ª—é—á–µ–≤—ã–º–∏ —Ç–æ—á–∫–∞–º–∏ –¥—Ä—É–≥–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è. BRIEF –≥–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –∫—Ä–∞—Ç–∫–æ–µ –±–∏–Ω–∞—Ä–Ω–æ–µ –æ–ø–∏—Å–∞–Ω–∏–µ —Ç–æ—á–µ–∫, –Ω–æ –Ω–µ —É—Å—Ç–æ–π—á–∏–≤ –∫ –ø–æ–≤–æ—Ä–æ—Ç—É –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π. –ó–¥–µ—Å—å –Ω–∞ –ø–æ–º–æ—â—å –ø—Ä–∏—Ö–æ–¥–∏—Ç "–ø–æ–≤–µ—Ä–Ω—É—Ç–∞—è" —á–∞—Å—Ç—å - ORB –¥–æ–±–∞–≤–ª—è–µ—Ç –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å —Å—Ç–∞–±–∏–ª—å–Ω–æ –æ–ø–∏—Å—ã–≤–∞—Ç—å —Ç–æ—á–∫–∏ –¥–∞–∂–µ –ø—Ä–∏ –ø–æ–≤–æ—Ä–æ—Ç–∞—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π.
